
dg_add_religion_modifiers_tolerance = {
	set_variable = { which = owner_religion_tolerance		value = 0 }
	set_variable = { which = dg_accepted_threshold			value = 3 }
	set_variable = { which = dg_tolerated_threshold			value = 1 }
	set_variable = { which = dg_persecuted_threshold		value = -1 }
	
	if = {
		limit = {
			check_variable = { which = toleration 	value = 1 }
		}
		subtract_variable = { which = dg_accepted_threshold 	which = toleration }
		subtract_variable = { which = dg_tolerated_threshold 	which = toleration }
		subtract_variable = { which = dg_persecuted_threshold 	which = toleration }
	}
	
	export_to_variable = {
		which = owner_religion_tolerance
		value = modifier:$selected_religion$
	}
	
	if = { limit = { check_variable = { which = $selected_religion$ value = 6 } }
		if = { limit = { check_variable = { which = $selected_religion$ value = 8 } }
			if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_tolerated_threshold } }
				if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_accepted_threshold } }
					add_permanent_province_modifier = { name = $selected_religion$_minority_8_accepted duration = -1 }
				}
				else = {
					add_permanent_province_modifier = { name = $selected_religion$_minority_8_tolerated duration = -1 }
				}
			}
			else = {
				if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_persecuted_threshold } }
					add_permanent_province_modifier = { name = $selected_religion$_minority_8_persecuted duration = -1 }
				}
				else = {
					add_permanent_province_modifier = { name = $selected_religion$_minority_8_oppressed duration = -1 }
				}
			}
		}
		else = {
			if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_tolerated_threshold } }
				if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_accepted_threshold } }
					add_permanent_province_modifier = { name = $selected_religion$_minority_6_accepted duration = -1 }
				}
				else = {
					add_permanent_province_modifier = { name = $selected_religion$_minority_6_tolerated duration = -1 }
				}
			}
			else = {
				if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_persecuted_threshold } }
					add_permanent_province_modifier = { name = $selected_religion$_minority_6_persecuted duration = -1 }
				}
				else = {
					add_permanent_province_modifier = { name = $selected_religion$_minority_6_oppressed duration = -1 }
				}
			}
		}
	}
	else = {
		if = { limit = { check_variable = { which = $selected_religion$ value = 4 } }
			if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_tolerated_threshold } }
				if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_accepted_threshold } }
					add_permanent_province_modifier = { name = $selected_religion$_minority_4_accepted duration = -1 }
				}
				else = {
					add_permanent_province_modifier = { name = $selected_religion$_minority_4_tolerated duration = -1 }
				}
			}
			else = {
				if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_persecuted_threshold } }
					add_permanent_province_modifier = { name = $selected_religion$_minority_4_persecuted duration = -1 }
				}
				else = {
					add_permanent_province_modifier = { name = $selected_religion$_minority_4_oppressed duration = -1 }
				}
			}
		}
		else = {
			if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_tolerated_threshold } }
				if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_accepted_threshold } }
					add_permanent_province_modifier = { name = $selected_religion$_minority_2_accepted duration = -1 }
				}
				else = {
					add_permanent_province_modifier = { name = $selected_religion$_minority_2_tolerated duration = -1 }
				}
			}
			else = {
				if = { limit = { check_variable = { which = owner_religion_tolerance which = dg_persecuted_threshold } }
					add_permanent_province_modifier = { name = $selected_religion$_minority_2_persecuted duration = -1 }
				}
				else = {
					add_permanent_province_modifier = { name = $selected_religion$_minority_2_oppressed duration = -1 }
				}
			}
		}
	}
	
	set_variable = { which = owner_religion_tolerance		value = 0 }
	set_variable = { which = dg_accepted_threshold			value = 0 }
	set_variable = { which = dg_tolerated_threshold			value = 0 }
	set_variable = { which = dg_persecuted_threshold		value = 0 }
}

dg_remove_religious_modifiers = {

		on_trigger = has_province_modifier

			remove_province_modifier = $religion$_minority_2_oppressed
		}
		$religion$_minority_4_oppressed = {
			remove_province_modifier = $religion$_minority_4_oppressed
		}
		$religion$_minority_6_oppressed = {
			remove_province_modifier = $religion$_minority_6_oppressed
		}
		$religion$_minority_8_oppressed = {
			remove_province_modifier = $religion$_minority_8_oppressed
		}
		$religion$_minority_2_persecuted = {
			remove_province_modifier = $religion$_minority_2_persecuted
		}
		$religion$_minority_4_persecuted = {

		}
		$religion$_minority_6_persecuted = {
			remove_province_modifier = $religion$_minority_6_persecuted
		}

dg_remove_heretic_modifiers = {
	trigger_switch = {
		on_trigger = has_province_modifier
		heretic_minority_2 = {
			remove_province_modifier = heretic_minority_2
		}
		heretic_minority_4 = {
			remove_province_modifier = heretic_minority_4
		}
		heretic_minority_6 = {
			remove_province_modifier = heretic_minority_6
		}
		heretic_minority_8 = {
			remove_province_modifier = heretic_minority_8
		}
	}
}

dg_assign_religious_modifiers = {
	clr_province_flag = heretic_presence
	#Heretics are always persecuted
	if = {
		limit = { check_variable = { which = heretics value = 2 } }
		if = { limit = { check_variable = { which = heretics value = 6 } }
			if = { limit = { check_variable = { which = heretics value = 8 } }
				if = { limit = { NOT = { has_province_modifier = heretic_minority_8 } }
					dg_remove_heretic_modifiers = yes
					add_permanent_province_modifier = { name = heretic_minority_8 duration = -1 }
				}
				set_province_flag = heretic_presence
			}
			else = {
				if = { limit = { NOT = { has_province_modifier = heretic_minority_6 } }
					dg_remove_heretic_modifiers = yes
					add_permanent_province_modifier = { name = heretic_minority_6 duration = -1 }
				}
				set_province_flag = heretic_presence
			}
		}
		else = {
			if = { limit = { check_variable = { which = heretics value = 4 } }
				if = { limit = { NOT = { has_province_modifier = heretic_minority_4 } }
					dg_remove_heretic_modifiers = yes
					add_permanent_province_modifier = { name = heretic_minority_4 duration = -1 }
				}
				set_province_flag = heretic_presence
			}
			else = {
				if = { limit = { NOT = { has_province_modifier = heretic_minority_2 } }
					dg_remove_heretic_modifiers = yes
					add_permanent_province_modifier = { name = heretic_minority_2 duration = -1 }
				}
				set_province_flag = heretic_presence
			}
		}
	}
	else = {
		dg_remove_heretic_modifiers = yes
	}
	if = {
		limit = { check_variable = { which = catholic value = 2 } }
		if = {
			limit = {
				owner = { NOT = { religion = catholic } }
				owner = { NOT = { has_country_flag = useful_catholic } }
				OR = {
					owner = { NOT = { has_country_flag = christians_permitted } }
					owner = { has_country_flag = chinese_rites_banned }
				}
			}
			set_province_flag = heretic_presence
		}
		if = {
			limit = {
				owner = {
					has_country_flag = christians_permitted
					NOT = { has_country_flag = chinese_rites_banned }
				}
			}
			set_variable = { which = toleration value = 1 }
		}
		if = {
			limit = {
				owner = {
					has_country_flag = christians_tolerated
					NOT = { has_country_flag = chinese_rites_banned }
				}
			}
			set_variable = { which = toleration value = 2 }
		}
		if = {
			limit = { owner = { religion = catholic } }
			set_variable = { which = toleration value = 0 }
		}
		if = {
			limit = {
				NOT = { religion = catholic }
			}
			if = { limit = { check_variable = { which = toleration value = 2 } }
				if = { limit = { check_variable = { which = toleration value = 3 } }
					dg_add_religion_modifiers = {
						selected_religion = catholic
						accepted_amount = 0
						tolerated_amount = -2
						persecuted_amount = -4
					}
				}
				else = {
					dg_add_religion_modifiers = {
						selected_religion = catholic
						accepted_amount = 1
						tolerated_amount = -1
						persecuted_amount = -3
					}
				}
			}
			else = {
				if = { limit = { check_variable = { which = toleration value = 1 } }
					dg_add_religion_modifiers = {
						selected_religion = catholic
						accepted_amount = 2
						tolerated_amount = 0
						persecuted_amount = -2
					}
				}
				else = {
					dg_add_religion_modifiers = {
						selected_religion = catholic
						accepted_amount = 3
						tolerated_amount = 1
						persecuted_amount = -1
					}
				}
			}
		}
		else = {
			dg_remove_religious_modifiers = {
				religion = catholic
			}
		}
		set_variable = { which = toleration value = 0 }
	}
	else = {
		dg_remove_religious_modifiers = {
			religion = catholic
		}
	}
	if = {
		limit = { check_variable = { which = orthodox value = 2 } }
		if = {
			limit = {
				owner = { NOT = { religion = orthodox } }
				owner = { NOT = { has_country_flag = useful_orthodox } }
				owner = { NOT = { has_country_flag = christians_permitted } }
				owner = { NOT = { has_country_flag = christians_tolerated } }
			}
			set_province_flag = heretic_presence
		}
		if = {
			limit = { owner = { has_country_flag = christians_permitted } }
			set_variable = { which = toleration value = 1 }
		}
		if = {
			limit = { owner = { has_country_flag = christians_tolerated } }
			set_variable = { which = toleration value = 2 }
		}
		if = {
			limit = {
				OR = {
					owner = { any_owned_province = { has_province_modifier = orthodox_patriarchate } }
					owner = { any_owned_province = { has_province_modifier = orthodox_patriarchate_captive } }
				}
			}
			change_variable = { which = toleration value = 1 }
		}
		if = {
			limit = { owner = { religion = orthodox } }
			set_variable = { which = toleration value = 0 }
		}
		if = {
			limit = {
				NOT = { religion = orthodox }
			}
			if = { limit = { check_variable = { which = toleration value = 2 } }
				if = { limit = { check_variable = { which = toleration value = 3 } }
					dg_add_religion_modifiers = {
						selected_religion = orthodox
						accepted_amount = 0
						tolerated_amount = -2
						persecuted_amount = -4
					}
				}
				else = {
					dg_add_religion_modifiers = {
						selected_religion = orthodox
						accepted_amount = 1
						tolerated_amount = -1
						persecuted_amount = -3
					}
				}
			}
			else = {
				if = { limit = { check_variable = { which = toleration value = 1 } }
					dg_add_religion_modifiers = {
						selected_religion = orthodox
						accepted_amount = 2
						tolerated_amount = 0
						persecuted_amount = -2
					}
				}
				else = {
					dg_add_religion_modifiers = {
						selected_religion = orthodox
						accepted_amount = 3
						tolerated_amount = 1
						persecuted_amount = -1

	else = {

			religion = orthodox

	}
	if = {
		limit = { check_variable = { which = coptic value = 2 } }
		if = {
			limit = {
				owner = { NOT = { religion = coptic } }
				owner = { NOT = { has_country_flag = useful_coptic } }
				owner = { NOT = { has_country_flag = christians_permitted } }
				owner = { NOT = { has_country_flag = christians_tolerated } }
			}
			set_province_flag = heretic_presence
		}
		if = {
			limit = { owner = { has_country_flag = christians_permitted } }
			set_variable = { which = toleration value = 1 }
		}
		if = {
			limit = { owner = { has_country_flag = christians_tolerated } }
			set_variable = { which = toleration value = 2 }
		}
		if = {
			limit = {
				OR = {
					owner = { any_owned_province = { has_province_modifier = coptic_patriarchate } }
					owner = { any_owned_province = { has_province_modifier = coptic_patriarchate_captive } }
				}
			}
			change_variable = { which = toleration value = 1 }
		}
		if = {
			limit = { owner = { religion = coptic } }
			set_variable = { which = toleration value = 0 }
		}
		if = {
			limit = {
				NOT = { religion = coptic }
			}
			if = { limit = { check_variable = { which = toleration value = 2 } }
				if = { limit = { check_variable = { which = toleration value = 3 } }
					dg_add_religion_modifiers = {
						selected_religion = coptic
						accepted_amount = 0
						tolerated_amount = -2
						persecuted_amount = -4
					}
				}
				else = {

dg_check_religious_turmoil = {
	if = {
		limit = {
			owner = { has_country_flag = religious_turmoil_$religion$ }
		}
		if = { limit = { check_variable = { which = $religion$ value = 12 } }
			if = { limit = { check_variable = { which = $religion$ value = 16 } }
				if = { limit = { check_variable = { which = $religion$ value = 20 } }
					change_variable = { which = religious_uprising value = 5 }
				}
				else = {
					change_variable = { which = religious_uprising value = 4 }
				}
			}
			else = {
				change_variable = { which = religious_uprising value = 3 }
			}
		}
		else = {
			if = { limit = { check_variable = { which = $religion$ value = 4 } }
				if = { limit = { check_variable = { which = $religion$ value = 8 } }
					change_variable = { which = religious_uprising value = 2 }
				}
				else = {
					change_variable = { which = religious_uprising value = 1 }
				}
			}
		}
	}
}
