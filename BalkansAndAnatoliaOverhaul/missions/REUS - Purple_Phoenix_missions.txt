	type = country
	
	category = MIL

		aydin_area
	}

		NOT = { owned_by = ROOT }
		NOT = { owner = { is_subject_of = ROOT } }
		is_empty = no
	}

		OR = { tag = BYZ tag = TRE }
		has_dlc = "Purple Phoenix"

		is_free_or_tributary_trigger = yes
		is_at_war = no
		total_development = 100
		num_of_owned_provinces_with = {
			OR = {
				#This region
				area = manisa_area
				area = aydin_area

				any_neighbor_province = { area = manisa_area }
				any_neighbor_province = { area = aydin_area }
				#Eastern med ports
				area = aegean_area
			}
			value = 2
		}
		OR = {
			NOT = { manisa_area = { type = all country_or_vassal_holds = ROOT } }

		is_subject_other_than_tributary_trigger = yes
	}

		all_target_province = {
			OR = {
				owned_by = ROOT

			}
		}
	}
	chance = {
		factor = 2400 #Highest factor
		modifier = {
			total_development = 300
			factor = 1.4
		}
		modifier = {
			total_development = 900
			factor = 1.4
		}
		modifier = {
			army_tradition = 75
			factor = 1.4
		}
		modifier = {
			army_tradition = 50
			factor = 1.4
		}
		modifier = {
			army_size_percentage = 1
			factor = 1.4
		}
		modifier = {
			army_size_percentage = 0.75
			factor = 1.4
		}

			factor = 1.4
