province_event = { #Monasticism
	
	id = dg_orthodox.001
	title = dg_orthodox.001.t
	desc = dg_orthodox.001.d
	picture = orthodox_monks
	
	is_triggered_only = yes #By maintenance event
	
	immediate = {
		set_province_flag = lavra #Settled areas, major cities
		set_province_flag = anchorite_hermitage #Only one province
		set_province_flag = cenobitic_monastery #Wilderness, colonies
		every_neighbor_province = {
			every_neighbor_province = {
				every_neighbor_province = {
					limit = {
						owned_by = ROOT
						NOT = { town_province_trigger = yes }
						NOT = { has_province_modifier = stavropegic_monastery }
						OR = {
							religion = orthodox
							is_colony = yes #Active in Africa and Brazil
							has_empty_adjacent_province = yes
						}
					}
					set_province_flag = cenobitic_monastery
				}
			}
		}
	}
	
	option = {
		name = dg_orthodox.001.a #Cenobites
		trigger = {
			OR = {
				has_owner_religion = yes
				is_colony = yes
			}
		}
		ai_chance = {
			factor = 40
		}
		owner = {
			every_owned_province = {
				limit = { has_province_flag = cenobitic_monastery }
				add_province_modifier = { name = cenobitic_monastery duration = 10000 }
				if = {
					limit = { NOT = { religion_group = christian } }
					custom_tooltip = orthodox_10
					hidden_effect = {
						change_variable = { which = orthodox value = 2 }
						change_variable = { which = province_religion value = -2 }
					}
				}
			}

	desc = dg_orthodox.081.d

		any_owned_province = { has_province_modifier = orthodox_patriarchate_captive }

			has_country_flag = religious_turmoil_orthodox
			any_owned_province = {
				has_missionary = yes
				religion = orthodox
				NOT = { has_province_modifier = missionary_failure }
			}

		NOT = { has_country_modifier = orthodox_patriarch_timer }
		NOT = { has_disaster = religious_civil_war }

					limit = { region = PREV }
					if = {

					if = {

						PREV = { change_variable = { which = patriarch_influence value = 1 } }
					}
				}
			}
		}
	}
	
	option = {
